\begin{defn}\label{D1.1}
	设 $ x,y \in \mathbb{N} $，自然数的加法用符号“+”表示，且满足：
	\[ \left\{ \begin{array}{l}
		x+0 = x, \\
		x+y' = (x+y)'. \\
	\end{array} \right. \]
\end{defn}

\begin{thm}[定理名]
	$ 1+1 = 2 $.
\end{thm}

\begin{proof}
	由定义 \ref{D1.1}，
	$ 1+1 = 1+0' = (1+0)' = 1' = 2 $.
\end{proof}
